public class ServicioGrupo2 {
    public static void validarTotalPedido(List<Producto> productos, double descuento){
        if(verificarStock(productos)) Pedido.calcularTotalPedido(productos, descuento);
        else throw new IllegalArgumentException("Error en funci√≥n secundaria");
    }
    
    public static boolean verificarStock(List<Producto> productos){
        return productos.stream()
            .anyMatch(p -> p.getCantidad() > 0);
    }
}
